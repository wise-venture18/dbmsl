-- Block (paste directly in SQL*Plus)
SET SERVEROUTPUT ON;
DECLARE
  v_roll  Stud.Roll%TYPE := &ROLL;
  v_att   Stud.Att%TYPE;
BEGIN
  SELECT Att INTO v_att FROM Stud WHERE Roll=v_roll;
  IF v_att < 75 THEN
    DBMS_OUTPUT.PUT_LINE('Term not granted');
    UPDATE Stud SET Status='D' WHERE Roll=v_roll;
  ELSE
    DBMS_OUTPUT.PUT_LINE('Term granted');
    UPDATE Stud SET Status='ND' WHERE Roll=v_roll;
  END IF;
EXCEPTION
  WHEN NO_DATA_FOUND THEN DBMS_OUTPUT.PUT_LINE('Roll not found');
END;
/
SELECT * FROM Stud WHERE Roll=&ROLL;
